00044 // Define system executor bit map. Used internally by runtime protocol as runtime command flags, 
00045 // which notifies the main program to execute the specified runtime command asynchronously.
00046 // NOTE: The system executor uses an unsigned 8-bit volatile variable (8 flag limit.) The default
00047 // flags are always false, so the runtime protocol only needs to check for a non-zero value to 
00048 // know when there is a runtime command to execute.
00049 #define EXEC_STATUS_REPORT  bit(0) // bitmask 00000001
00050 #define EXEC_CYCLE_START    bit(1) // bitmask 00000010
00051 #define EXEC_CYCLE_STOP     bit(2) // bitmask 00000100
00052 #define EXEC_FEED_HOLD      bit(3) // bitmask 00001000
00053 #define EXEC_RESET          bit(4) // bitmask 00010000
00054 #define EXEC_ALARM          bit(5) // bitmask 00100000
00055 #define EXEC_CRIT_EVENT     bit(6) // bitmask 01000000
00056 // #define                  bit(7) // bitmask 10000000
00057 
00058 // Define system state bit map. The state variable primarily tracks the individual functions
00059 // of Grbl to manage each without overlapping. It is also used as a messaging flag for
00060 // critical events.
00061 #define STATE_IDLE       0      // Must be zero. No flags.
00062 #define STATE_ALARM      bit(0) // In alarm state. Locks out all g-code processes. Allows settings access.
00063 #define STATE_CHECK_MODE bit(1) // G-code check mode. Locks out planner and motion only.
00064 #define STATE_HOMING     bit(2) // Performing homing cycle
00065 #define STATE_QUEUED     bit(3) // Indicates buffered blocks, awaiting cycle start.
00066 #define STATE_CYCLE      bit(4) // Cycle is running
00067 #define STATE_HOLD       bit(5) // Executing feed hold
00068 // #define STATE_JOG     bit(6) // Jogging mode is unique like homing.
