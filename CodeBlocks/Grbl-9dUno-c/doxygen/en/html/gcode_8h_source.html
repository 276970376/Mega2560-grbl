00027 // Define modal group internal numbers for checking multiple command violations and tracking the 
00028 // type of command that is called in the block. A modal group is a group of g-code commands that are
00029 // mutually exclusive, or cannot exist on the same line, because they each toggle a state or execute
00030 // a unique motion. These are defined in the NIST RS274-NGC v3 g-code standard, available online, 
00031 // and are similar/identical to other g-code interpreters by manufacturers (Haas,Fanuc,Mazak,etc).
00032 #define MODAL_GROUP_NONE 0
00033 #define MODAL_GROUP_0 1 // [G4,G10,G28,G30,G53,G92,G92.1] Non-modal
00034 #define MODAL_GROUP_1 2 // [G0,G1,G2,G3,G38.2,G80] Motion
00035 #define MODAL_GROUP_2 3 // [G17,G18,G19] Plane selection
00036 #define MODAL_GROUP_3 4 // [G90,G91] Distance mode
00037 #define MODAL_GROUP_4 5 // [M0,M1,M2,M30] Stopping
00038 #define MODAL_GROUP_5 6 // [G93,G94] Feed rate mode
00039 #define MODAL_GROUP_6 7 // [G20,G21] Units
00040 #define MODAL_GROUP_7 8 // [M3,M4,M5] Spindle turning
00041 #define MODAL_GROUP_8 9 // [M7,M8,M9] Coolant control
00042 #define MODAL_GROUP_12 10 // [G54,G55,G56,G57,G58,G59] Coordinate system selection
